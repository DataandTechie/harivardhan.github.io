import pandas as pd
from sklearn.cluster import KMeans

# Load data
data = pd.read_csv('customers.csv')

# Preprocessing
X = data[['Age', 'Income', 'SpendingScore']]

# Apply K-means
kmeans = KMeans(n_clusters=3)
kmeans.fit(X)
data['Cluster'] = kmeans.labels_
